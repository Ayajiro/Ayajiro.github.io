# Port被佔用的情況

%% 列出所有監聽中的port %%
netstat -aon | find /i "listening"

%% 指定搜尋被佔用的port %%
netstat -ano | findstr :3000

%% 依剛才搜尋port取得PID 清除指定PID %%
taskkill /F /PID [PID]

也可以安裝 kill-port工具刪除指定port

npx kill-port 3000 3001 8000
